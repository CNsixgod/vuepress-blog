## 二分搜索

```js
Array.protortype.quickSort = function(item) {
	const low = 0
  const high = this.length - 1
  
  while(low < high) {
    const mid = Math.floor((low + high) / 2)
    const ele = this[mid]
    if(ele < item) {
      low = mid + 1
    } else if (ele > item) {
      high = mid - 1
    } else {
      return mid
    }
  }
  return -1
}

```
