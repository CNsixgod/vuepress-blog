    用setTimeout模拟setInterval

    2/12/2021 
    function mySetTimeout(fn, t) {
  let timer = null
  function interval() {
    timer = setTimeout(() => {
      clearTimeout(timer)
      fn()
      interval()
    }, t)
  }
  interval()
  return {
    cancel: () => {
      clearTimeout(timer)
    },
  }
}

  console.log(111)
